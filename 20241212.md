## 오늘 백준 무제푼거


#include <iostream>
#include<cmath>
using namespace std;

int main(int argc, const char * argv[]) {
    // insert code here...
    ios_base::sync_with_stdio(false);
    
    int n,b;
    string res;
    cin>>n>>b;
    
    while(n>0){
        int tmp = n%b;
        if(0<= tmp && tmp<=9)
            res += (tmp + '0');
        else
            res += (tmp-10 +'A');
        n/=b;
        
    }
    
    for(int i = res.length()-1 ; i>=0; i--){
        cout<< res[i];
        // 거꾸로 출력하기
    }
}
